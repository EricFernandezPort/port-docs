          identifier: .objectId
          title: .path
          url: .__repository.url
          blueprint: '"folder"'

            description: .__repository.project.description
            project: .__repository.project.name
            repo: .__repository.name
            path: .path
            branch: .__branch
          relations:
            repository: >-
              .__repository.project.name + "/" + .__repository.name | gsub(" ";
